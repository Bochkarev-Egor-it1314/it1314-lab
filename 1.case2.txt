#include <iostream>
using namespace std;

int main()
{
    setlocale(LC_ALL, "RU");

    int k;
    cout << "Введите число от 1 до 5: " << endl;
    cin >> k;

    switch (k)
    {
        case 1: cout << "Плохо"; break;
        case 2: cout << "Неудавлетворительно"; break;
        case 3: cout << "Удовлетворительно"; break;
        case 4: cout << "Хорошо"; break;
        case 5: cout << "Отлично"; break;
        default: cout << "ОШИБКА"; break;
    }


}
